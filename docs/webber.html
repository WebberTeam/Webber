 78        Base class used to execute DAG in embarrassingly parallel.

 80        def __init__(self, graph: _nx.DiGraph, roots: list) -> None:
 81            with _OutputLogger(str(_uuid.uuid1()), "INFO", "root") as _:
 82                # Initialize local variables for execution.
 83                complete, started, failed, skipped = set(), set(), set(), set()
 84                events = set(roots)
 85                refs = {}
 86                # Start execution of root node functions.
 87                with _futures.ThreadPoolExecutor() as executor:
 88                    for event in events:
 89                        refs[event] = executor.submit(
 90                            _event_wrapper,
 91                            _callable=graph.nodes[event]['callable'],
 92                            _name=graph.nodes[event]['name'],
 93                            _args=graph.nodes[event]['args'],
 94                            _kwargs=graph.nodes[event]['kwargs']
 95                        )
 96                        started.add(event)
 97                    # Loop until all nodes in the network are executed.
 98                    while (len(complete) + len(failed) + len(skipped)) != len(graph):
 99                        for event in events:
100                            if refs[event].done() is True:
101                                if refs[event].exception(timeout=0) is not None:
102                                    try:
103                                        raise refs[event].exception(timeout=0)
104                                    except:                                                                             # pylint: disable=bare-except
105                                        _traceback.print_exc()
106                                        print(f"Event {event} exited with exception, skipping dependent events...")     # pylint: disable=line-too-long
107                                        failed.add(event)
108                                        skipped = skipped.union(_nx.descendants(graph, event))
109                                        started = started.union(skipped)
110                                        events  = started.difference(complete.union(failed).union(skipped))             # pylint: disable=line-too-long
111                                        continue
112                                complete.add(event)
113                                successors = [
114                                    successor for successor in list(graph.successors(event)) if
115                                    complete.issuperset(set(graph.predecessors(successor)))
116                                ]
117                                for successor in successors:
118                                    _args = [
119                                        a if not isinstance(a, _xcoms.Promise) else refs[a.key].result()
120                                        for a in graph.nodes[successor]['args']
121                                    ]
122                                    _kwargs = {
123                                        k: v if not isinstance(v, _xcoms.Promise) else refs[v.key].result()
124                                        for k, v in graph.nodes[successor]['kwargs'].items()
125                                    }
126                                    refs[successor] = executor.submit(
127                                        _event_wrapper,
128                                        _callable=graph.nodes[successor]['callable'],
129                                        _name=graph.nodes[successor]['name'],
130                                        _args=_args,
131                                        _kwargs=_kwargs
132                                    )
133                                    started.add(successor)
134                        # Initialized nodes that are incomplete or not yet documented as complete.
135                        events = started.difference(complete.union(failed).union(skipped))
